"""Alpha factor definition — the ONLY file the agent may edit."""

from __future__ import annotations

import numpy as np
import pandas as pd


def compute_factor(
    symbols: list[str],
    as_of: pd.Timestamp,
    data: dict,
) -> dict[str, float]:
    """3-rank: rank(RevAccel4M)+2*rank(Amihud_30d)+rank(MA110_avg_dist).
    RevAccel4M: 4M/8M revenue ratio. Amihud 30d double-weighted.
    MA110: 63d avg SMA distance.
    """
    raw: dict[str, tuple] = {}

    for sym in symbols:
        try:
            rev = data["revenue"].get(sym)
            if rev is None or rev.empty: continue
            r = rev.sort_values("date")
            if len(r) < 12: continue
            vals = r["revenue"].values[-12:]
            base = vals[:-4].mean()
            if base <= 0: continue
            accel = vals[-4:].mean() / base - 1.0

            bars = data["bars"].get(sym)
            if bars is None or bars.empty: continue
            b = bars.loc[:as_of]
            c = b["close"].values
            if len(c) < 163: continue
            close30 = c[-31:]; vol = b["volume"].values[-30:]
            ret_abs = np.abs(np.diff(close30) / close30[:-1])
            dvol = close30[1:] * vol / 1e6
            if dvol.min() <= 0: continue
            amihud = np.mean(ret_abs / dvol)
            ma_d = np.mean([c[-i-1]/c[-i-111:-i-1].mean()-1 for i in range(63)])
            raw[sym] = (accel, amihud, ma_d)
        except Exception:
            continue

    if not raw:
        return {}
    syms = list(raw.keys())
    rv = np.array([raw[s][0] for s in syms])
    mi = np.array([raw[s][1] for s in syms])
    ma = np.array([raw[s][2] for s in syms])
    n = len(syms)
    def rk(x): return np.argsort(np.argsort(x)) / n
    return {s: float(rk(rv)[i] + 2*rk(mi)[i] + rk(ma)[i]) for i, s in enumerate(syms)}
